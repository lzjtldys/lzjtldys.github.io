def excrt(m, r):
   lcm = m[0]
   R = r[0]
   for i in range(1, len(m)):
       d, x, y = gcdext(lcm, m[i])
       c = R - r[i]
       if c % d:
           return -1
       x = x * c // d % (m[i] // d)
       R -= lcm * x
       lcm = lcm * m[i] // d
       R %= lcm
   return (R + lcm) % lcm
